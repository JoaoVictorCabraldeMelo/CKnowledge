#include <bits/stdc++.h>

  

using namespace std;

  

#define vi vector<int>

#define ll long long

#define pb push_back

#define mp make_pair

#define ii pair<int, int>

#define nl "\n"

  

int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n, m;
	cin >> n >> m;
	queue<ii> students;
	for (int i = 0; i < n; i++) {
		int x;
		cin >> x;
		ii student = {i + 1, x};
		students.push(student);
	}

	while (students.size() > 1) {
		if (students.front().second <= m)
			students.pop();
		else {
			ii put_back = students.front();
			put_back.second -= m;
			students.push(put_back);
			students.pop();
		}
	}
	cout << students.front().first << nl;
	return 0;
}
